SQL语句

SQL主要由六个子语言组成，分别是 DDL、DQL、DML、DCL、TCL（TPL）和 CCL

DDL（Data Definition Language，数据定义语言）

xxxxxxxxxx
create - 创建
alter - 修改
drop - 删除
TRUNCATE - 截断
COMMENT - 注释
RENAME - 重命名

DQL（Data Query Language，数据查询语言）

xxxxxxxxxx
select-从数据库表中获取数据 
from - 指定从哪个数据表或者子查询中查询
where - 指定查询条件
GROUP BY - 结合合计函数，根据一个或多个列对结果集进行分组
HAVING - 对分组后的结果集进行筛选
ORDER BY - 对结果集进行排序
LIMIT - 对结果集进行top限制输出
UNION - 结果集纵向联合
JOIN - 结果集横向拼接

DML（Data Manipulation Language，数据操作语言）

xxxxxxxxxx
update - 更新数据库表中的数据
delete - 从数据库表中删除数据
insert into - 向数据库表中插入数据
REPLACE INTO- 向数据库表中插入数据，如果存在先删除
LOAD - 载入数据

表

建表

xxxxxxxxxx
#结构
create table [表名] ( [字段名1] [类型1] [is null] [key] [default value] [extra] [comment],
...
)[engine] [charset];
​
#例
create table 表(
    可借天数 int    #类别
    类别号 char(2) primary key,    #主键
    类名 char(10) not null,   #非空
    可借数量 int default '',    #默认值
    性别 char(2) check(性别 in ('男','女'));  #约束
);
​
​
#复制表
create table 新表名 like 旧表名;

删表

xxxxxxxxxx
drop table [tablename];     #删除表
delete from [tablename];    #清空表
truncate [tablename];       #清空表

改表

xxxxxxxxxx
#修改列属性
alter table 表 modify [columnname] [newdefinition];
#例如
alter table student modify home varchar(64) not null;
​
#增类型
alter table 表 add 字段名 类型;
#删类型
alter table 表 drop 字段名;
#改类型
alter table 表 change 旧字段名  新字段名 类型;
#增主键
alter table 表 add primary key(Field);
#删主键
alter table 表 drop primary key;
#增外键
alter table 表2 add foreign key(列) references 表1(列);
#删外键
alter table 从表名 drop foreign key 外键名;
#增约束
alter table 表 add check(Field in ('在馆','借出','丢失'));
alter table 表 add check('0'<Field<'30');
#删约束
​
#重命名表
rename table [tablename] to [newtablename];

查数据

xxxxxxxxxx
#完整的SELECT语句包含可选的几个子句
<SELECT clause> [FROM clause] [WHERE clause] [GROUP BY clause] [HAVING clause] [ORDER BY clause] [LIMIT clause]
​
#别名
select 姓名 as name,单位 as college from 表;
​
#替换
select 条码,
case
when 库存状态='在馆' then '1'
when 库存状态='借出' then '0'
when 库存状态='丢失' then '-1'
end
库存状态
from 表;
​
select
case
when 成绩 >=90 then ''
when 成绩 between 75 and 90 then ''
when 成绩 between 60 and 75 then ''
else ''
end
成绩
from 表;
​
#计算
select 书名,数量*单价 as 金额 from 表;
select 姓名 as name,year(now())-year(出生日期) as age from 表;
​
##条件
#比较运算
select * from 表 where 学时 >= 48;
select * from 表 where 数量>10;
#逻辑运算
select * from 表 where 数量>5 and 类别 in ('财经','文学');
select * from 表 where (出版社='' or 出版社='') and 单价>='';
#模糊查询（‘_’代表单个字符）
select * from 表 where 位置 like'__A%' and 库存状态 like'借出';
#空值比较
select * from 表 where 前置课 is [not] null;
​
#多表
select * from 表1 join 表2 using();
select * from 表1 left join 表2 using();
select * from 表 where [] in (select * from 表 where []='');
​
#分类汇总
select count(*),max(),min(),sum(),avg() from 表 group by [] having 条件;
​
#排序
select * from 表 order by [];
select * from 表 order by [] desc;
select * from 表 limit 5;
